# Test with slashes
>>> from problem_7 import Router
>>> router = Router("root handler", "not found handler")
>>> router.add_handler("/home/about", "about handler")
>>> router.lookup("/")
'root handler'
>>> router.lookup("/home")
'not found handler'
>>> router.lookup("/home/about")
'about handler'
>>> router.lookup("//home//about/")
'about handler'
>>> router.lookup("/home/about/")
'about handler'
>>> router.lookup("/home/about/me")
'not found handler'
>>> router.lookup("/home//about/me/")
'not found handler'
>>> router.lookup("/about")
'not found handler'
>>> router.lookup("/about/")
'not found handler'
